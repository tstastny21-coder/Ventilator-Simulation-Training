<script>

(function(){

  "use strict";


  // ---------- state ----------

  const cfg = { R:8, C:20, Vt:300, Ti:1.0, RR:14, PEEP:3, FiO2:40 };

  const sim = {

    running:false,

    V_L:0,            // volume above PEEP baseline, litres

    tInBreath:0,      // seconds since start of current breath

    clock:0,          // total elapsed sim time, seconds

    buffer:[],        // {t,P,F,V}

    curPpeak:0,

    lastVte:null,

    breathVpeak:0,

    lastTrapped_mL:0

  };

  const WINDOW_S = 10;      // seconds visible on scope

  const DT = 1/120;         // integration step (s)


  const presets = {

    dog_normal:      {R:8,  C:25},

    cat_normal:      {R:12, C:9},

    cat_obstructive: {R:30, C:9},

    dog_restrictive: {R:9,  C:8},

    dog_obstructive: {R:22, C:22}

  };


  // ---------- element refs ----------

  const el = id => document.getElementById(id);

  const inputs = ['R','C','Vt','Ti','RR','PEEP','FiO2'];

  const suffix = {R:' cmH₂O/L/s', C:' mL/cmH₂O', Vt:' mL', Ti:' s', RR:' /min', PEEP:' cmH₂O', FiO2:' %'};


  function syncLabels(){

    el('vR').textContent    = cfg.R.toFixed(1) + suffix.R;

    el('vC').textContent    = cfg.C.toFixed(0) + suffix.C;

    el('vVt').textContent   = cfg.Vt.toFixed(0) + suffix.Vt;

    el('vTi').textContent   = cfg.Ti.toFixed(2) + suffix.Ti;

    el('vRR').textContent   = cfg.RR.toFixed(0) + suffix.RR;

    el('vPEEP').textContent = cfg.PEEP.toFixed(0) + suffix.PEEP;

    el('vFiO2').textContent = cfg.FiO2.toFixed(0) + suffix.FiO2;

  }


  inputs.forEach(k=>{

    el(k).addEventListener('input', e=>{

      cfg[k] = parseFloat(e.target.value);

      if(k==='R' || k==='C') el('presetSelect').value = 'custom';

      syncLabels();

      recomputeAxes();

    });

  });


  el('presetSelect').addEventListener('change', e=>{

    const p = presets[e.target.value];

    if(!p) return;

    cfg.R = p.R; cfg.C = p.C;

    el('R').value = p.R; el('C').value = p.C;

    syncLabels();

    recomputeAxes();

  });


  // ---------- derived quantities ----------

  let tau, period, Te, VtL, C_L, insFlow_Ls;

  let pMax, fMax, vMax;


  function recomputeDerived(){

    C_L = cfg.C/1000;

    tau = cfg.R * C_L;                     // seconds

    period = 60/cfg.RR;

    Te = Math.max(0.05, period - cfg.Ti);

    VtL = cfg.Vt/1000;

    insFlow_Ls = VtL/cfg.Ti;

  }

  function recomputeAxes(){

    recomputeDerived();

    const estPeakP = cfg.PEEP + (VtL/C_L) + cfg.R*insFlow_Ls;

    pMax = Math.max(40, estPeakP*1.25);

    const peakFlowLmin = insFlow_Ls*60;

    fMax = Math.max(40, peakFlowLmin*1.3);

    vMax = Math.max(150, cfg.Vt*1.5);

    el('roTau').innerHTML = tau.toFixed(2)+'<span class="unit">s</span>';

    const ie = (cfg.Ti/Te);

    el('roIE').textContent = '1 : ' + (1/ie).toFixed(1);

  }


  // ---------- simulation step ----------

  function step(dt){

    recomputeDerived();

    const insp = sim.tInBreath < cfg.Ti;

    let flow_Ls;

    if(insp){

      flow_Ls = insFlow_Ls;

    } else {

      flow_Ls = tau>0 ? -(sim.V_L/tau) : 0;

    }

    sim.V_L += flow_Ls*dt;

    if(sim.V_L < 0) sim.V_L = 0;


    const Paw = sim.V_L/C_L + cfg.R*flow_Ls + cfg.PEEP;


    if(insp){

      sim.breathVpeak = sim.V_L;

      sim.curPpeak = Paw;

    }


    sim.clock += dt;

    sim.tInBreath += dt;


    sim.buffer.push({t:sim.clock, P:Paw, F:flow_Ls*60, V:sim.V_L*1000});

    const cutoff = sim.clock - WINDOW_S - 0.5;

    while(sim.buffer.length && sim.buffer[0].t < cutoff) sim.buffer.shift();


    if(sim.tInBreath >= period){

      // breath boundary: whatever volume remains above baseline is trapped gas

      sim.lastTrapped_mL = sim.V_L*1000;

      sim.lastVte = (sim.breathVpeak - sim.V_L)*1000;

      sim.tInBreath -= period;

      updateReadouts();

    }

  }


  function updateReadouts(){

    el('roPpeak').innerHTML = sim.curPpeak.toFixed(1)+'<span class="unit">cmH₂O</span>';

    el('roVte').innerHTML = (sim.lastVte==null? '&mdash;' : sim.lastVte.toFixed(0))+'<span class="unit">mL</span>';

    const box = el('roAutoPeepBox');

    const trapped = sim.lastTrapped_mL||0;

    if(trapped > 8){

      const pTrap = (trapped/1000)/C_L;

      el('roAutoPeep').innerHTML = pTrap.toFixed(1)+'<span class="unit">cmH₂O</span>';

      box.classList.add('alert'); box.classList.remove('ok');

    } else {

      el('roAutoPeep').textContent = 'none';

      box.classList.remove('alert'); box.classList.add('ok');

    }

  }


  // ---------- rendering ----------

  const canvas = el('scope');

  const ctx = canvas.getContext('2d');

  function resizeCanvas(){

    const rect = canvas.getBoundingClientRect();

    const ratio = window.devicePixelRatio||1;

    canvas.width = rect.width*ratio;

    canvas.height = rect.height*ratio;

    ctx.setTransform(ratio,0,0,ratio,0,0);

  }

  window.addEventListener('resize', resizeCanvas);


  const bandLabels = [

    {key:'P', name:'Paw  cmH₂O', color:getVar('--pressure'), get max(){return pMax;}, zero:0},

    {key:'F', name:'Flow  L/min', color:getVar('--flow'), get max(){return fMax;}, zero:'mid'},

    {key:'V', name:'Volume  mL', color:getVar('--volume'), get max(){return vMax;}, zero:0}

  ];

  function getVar(name){return getComputedStyle(document.documentElement).getPropertyValue(name).trim();}


  function draw(){

    const w = canvas.clientWidth, h = canvas.clientHeight;

    ctx.clearRect(0,0,w,h);

    const gap = 10;

    const bandH = (h - gap*4)/3;


    bandLabels.forEach((band, i)=>{

      const top = gap + i*(bandH+gap);

      drawBand(band, top, bandH, w);

    });


    // playhead line

    ctx.strokeStyle = 'rgba(255,255,255,0.15)';

    ctx.beginPath();

    ctx.moveTo(w-2,0); ctx.lineTo(w-2,h);

    ctx.stroke();

  }


  function drawBand(band, top, bandH, w){

    // background + border

    ctx.fillStyle = getVar('--panel-2');

    ctx.fillRect(0, top, w, bandH);

    ctx.strokeStyle = getVar('--line-soft');

    ctx.strokeRect(0.5, top+0.5, w-1, bandH-1);


    // grid: horizontal

    ctx.strokeStyle = getVar('--line-soft');

    ctx.lineWidth = 1;

    for(let g=1; g<4; g++){

      const y = top + bandH*g/4;

      ctx.beginPath(); ctx.moveTo(0,y); ctx.lineTo(w,y); ctx.stroke();

    }

    // vertical time gridlines every 1s

    for(let s=0; s<=WINDOW_S; s++){

      const x = w - (s/WINDOW_S)*w;

      ctx.beginPath(); ctx.moveTo(x,top); ctx.lineTo(x,top+bandH); ctx.stroke();

    }


    // zero / baseline reference line

    const valToY = v=>{

      if(band.zero==='mid'){

        return top + bandH/2 - (v/band.max)*(bandH/2);

      }

      return top + bandH - (v/band.max)*bandH;

    };

    ctx.strokeStyle = 'rgba(255,255,255,0.18)';

    const zy = valToY(0);

    ctx.setLineDash([3,3]);

    ctx.beginPath(); ctx.moveTo(0,zy); ctx.lineTo(w,zy); ctx.stroke();

    ctx.setLineDash([]);


    // label

    ctx.fillStyle = getVar('--text-faint');

    ctx.font = '11px "IBM Plex Mono", monospace';

    ctx.fillText(band.name, 8, top+14);


    // trace

    if(sim.buffer.length > 1){

      ctx.beginPath();

      ctx.lineWidth = 1.8;

      ctx.strokeStyle = band.color;

      ctx.lineJoin = 'round';

      let started = false;

      for(let i=0;i<sim.buffer.length;i++){

        const s = sim.buffer[i];

        const x = w - (sim.clock - s.t)/WINDOW_S * w;

        if(x < -5) continue;

        const y = valToY(s[band.key]);

        if(!started){ ctx.moveTo(x,y); started = true; }

        else ctx.lineTo(x,y);

      }

      ctx.stroke();

    }

  }


  // ---------- main loop ----------

  let lastTs = null;

  function frame(ts){

    if(!sim.running){ lastTs = null; return; }

    if(lastTs==null) lastTs = ts;

    let elapsed = (ts-lastTs)/1000;

    lastTs = ts;

    elapsed = Math.min(elapsed, 0.1); // clamp huge gaps (tab switch etc)

    let remaining = elapsed;

    while(remaining > 0){

      const dt = Math.min(DT, remaining);

      step(dt);

      remaining -= dt;

    }

    draw();

    requestAnimationFrame(frame);

  }


  el('playBtn').addEventListener('click', ()=>{

    sim.running = !sim.running;

    el('playBtn').textContent = sim.running ? 'Pause' : 'Start';

    if(sim.running) requestAnimationFrame(frame);

  });


  el('resetBtn').addEventListener('click', ()=>{

    sim.running = false;

    el('playBtn').textContent = 'Start';

    sim.V_L = 0; sim.tInBreath = 0; sim.clock = 0; sim.buffer = [];

    sim.curPpeak = 0; sim.lastVte = null; sim.breathVpeak = 0; sim.lastTrapped_mL = 0;

    updateReadouts();

    resizeCanvas();

    draw();

  });


  // ---------- init ----------

  syncLabels();

  recomputeAxes();

  resizeCanvas();

  draw();

  updateReadouts();

})();

</script>
